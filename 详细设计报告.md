3.2.1 Result - 统一响应结果

@Data
public class Result<T> {
    private Integer code;  响应码：200成功，500失败
    private String msg;    响应消息
    private T data;        响应数据

    成功响应
    public static <T> Result<T> success(T data) {
        Result<T> result = new Result<>();
        result.setCode(200);
        result.setMsg("success");
        result.setData(data);
        return result;
    }

    失败响应
    public static <T> Result<T> error(String msg) {
        Result<T> result = new Result<>();
        result.setCode(500);
        result.setMsg(msg);
        return result;
    }
}

CREATE TABLE `storage` (

  PRIMARY KEY (`id`)

5.2.3 商品表 (goods)

CREATE TABLE `goods` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '商品ID',
  `name` varchar(100) NOT NULL COMMENT '商品名称',
  `storageId` int NOT NULL COMMENT '仓库ID',
  `storageName` varchar(100) NOT NULL COMMENT '仓库名称',
  `amount` int NOT NULL DEFAULT 0 COMMENT '库存数量',
  `barcode` varchar(50) DEFAULT NULL COMMENT '商品条码',

  PRIMARY KEY (`id`),
  KEY `idx_barcode` (`barcode`),
  KEY `idx_storage` (`storageId`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品表';

5.2.4 出入库记录表 (record)

CREATE TABLE `record` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '记录ID',
  `goodsId` int NOT NULL COMMENT '商品ID',
  `goodsName` varchar(100) NOT NULL COMMENT '商品名称',
  `storageId` int NOT NULL COMMENT '仓库ID',
  `storageName` varchar(100) NOT NULL COMMENT '仓库名称',
  `type` int NOT NULL COMMENT '操作类型 1入库 2出库',
  `amount` int NOT NULL COMMENT '变动数量',
  `time` datetime NOT NULL COMMENT '操作时间',
  PRIMARY KEY (`id`),
  KEY `idx_goods` (`goodsId`),
  KEY `idx_time` (`time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='出入库记录表';

5.2.5 库存预警表 (alert)

CREATE TABLE `alert` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '预警ID',
  `goodsId` int NOT NULL COMMENT '商品ID',
  `goodsName` varchar(100) NOT NULL COMMENT '商品名称',
  `amount` int NOT NULL COMMENT '当前库存',
  `isRead` tinyint(1) NOT NULL DEFAULT 0 COMMENT '是否已读 0未读 1已读',
  `time` datetime NOT NULL COMMENT '预警时间',
  PRIMARY KEY (`id`),
  KEY `idx_read` (`isRead`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='库存预警表';

六、关键业务流程设计

6.1 商品入库流程

用户选择商品和仓库 -> 验证数据(数量>0) -> 更新库存 -> 记录日志

后端实现：
@PostMapping("/inbound")
public Result<?> inbound(@RequestBody Map<String, Object> params) {
    Integer goodsId = (Integer) params.get("goodsId");
    Integer amount = (Integer) params.get("amount");

    Goods goods = goodsMapper.selectById(goodsId);
    goods.setAmount(goods.getAmount() + amount);
    goodsMapper.updateById(goods);

    记录入库日志
    Record record = new Record();
    record.setGoodsId(goodsId);
    record.setGoodsName(goods.getName());
    record.setStorageId(goods.getStorageId());
    record.setStorageName(goods.getStorageName());
    record.setType(1); // 入库
    record.setAmount(amount);
    record.setTime(new Date());
    recordMapper.insert(record);

    return Result.success(null);
}

6.2 商品出库流程

用户选择商品和数量 -> 验证数据(数量≤当前库存) -> 更新库存 -> 记录日志

6.3 库存检查流程

遍历所有商品 -> 检查库存是否<10 -> 创建预警记录 -> 保存到数据库

后端实现：
@PostMapping("/check")
public Result<?> checkAlert() {
    List<Goods> goodsList = goodsMapper.selectList(null);
    for (Goods goods : goodsList) {
        if (goods.getAmount() < 10) {
            查询是否已有预警
            QueryWrapper<Alert> wrapper = new QueryWrapper<>();
            wrapper.eq("goodsId", goods.getId()).eq("isRead", false);
            Alert existAlert = alertMapper.selectOne(wrapper);

            if (existAlert == null) {
                Alert alert = new Alert();
                alert.setGoodsId(goods.getId());
                alert.setGoodsName(goods.getName());
                alert.setAmount(goods.getAmount());
                alert.setIsRead(false);
                alert.setTime(new Date());
                alertMapper.insert(alert);
            }
        }
    }
    return Result.success(null);
}
